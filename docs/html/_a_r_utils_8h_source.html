92  // Set the dimensions as appropriate depending on our orientation.
93  // Note that for some reason, and I'm not sure if it's an oF, IOS or mistake on my part, but the first time
94  // it enters this switch block, dimensions are off, so there is an nested if statement to try and fix that in
95  // some of the cases.
96  switch(UIDevice.currentDevice.orientation){
97  case UIDeviceOrientationFaceUp:
98  setWidthAndHeight();
99

101  dimensions.x = width;
102  dimensions.y = height;
103  break;
104 
105  case UIDeviceOrientationFaceDown:

107 
108  // if face up - we just assume portrait

111  break;

113  // if unknown - we just assume portrait
114  dimensions.x = width;
115  dimensions.y = height;
116  break;
117 
118  // upside down registers, but for some reason nothing happens and there might be weirdness :/

201  // convert world xyz position to screen position.
202  // TODO maybe this is what we ought to use instead to factor in orientation? Found on 11/7/17
203  // https://developer.apple.com/documentation/arkit/arcamera/2923538-projectpoint?language=objc
204  static ofVec2f worldToScreen(ofPoint worldPoint,ofMatrix4x4 projection,ofMatrix4x4 view){
205 
206  ofVec4f p = ofVec4f(worldPoint.x, worldPoint.y, worldPoint.z, 1.0);
207 
208  p = p * view;
209  p = p * projection;
210 
211  p /= p.w;
212 
213  // convert coords to 0 - 1
214  p.x = p.x * 0.5 + 0.5;
215  p.y = p.y * 0.5 + 0.5;
216 
217  // convert coords to pixels
218  p.x *= ofGetWidth();
219  p.y *= ofGetHeight();
220  return ofVec2f(p.x, p.y);
221  }

224  static ofVec4f screenToWorld(ofVec3f position,ofMatrix4x4 projection,ofMatrix4x4 mvMatrix){
225  ofRectangle viewport(0,0,ofGetWindowWidth(),ofGetWindowHeight());
226 
227  ofVec4f CameraXYZ;
228  CameraXYZ.x = 2.0f * (position.x - viewport.x) / viewport.width - 1.0f;
229  CameraXYZ.y = 1.0f - 2.0f *(position.y - viewport.y) / viewport.height;
230  CameraXYZ.z = position.z;
231  CameraXYZ.w = -10.0;
232 
233  ofMatrix4x4 inverseCamera;
234  inverseCamera.makeInvertOf(mvMatrix * projection);
235 
236  return CameraXYZ * inverseCamera;
237  }
