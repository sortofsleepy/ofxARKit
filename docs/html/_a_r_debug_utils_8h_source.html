132  std::set_difference(uniqueIds.begin(), uniqueIds.end(),
133  oldSet.begin(), oldSet.end(),
134  std::inserter(difference, difference.end()));
135 
136  //if there is anything new put it in newpoints

144  vector_float3 p = pointCloud.points[i];
